def remover_acentos(texto):
    # Mapeamento de caracteres acentuados para não acentuados
    mapeamento_acentos = {
        'á': 'a', 'é': 'e', 'í': 'i', 'ó': 'o', 'ú': 'u',
        'â': 'a', 'ê': 'e', 'î': 'i', 'ô': 'o', 'û': 'u',
        'à': 'a', 'è': 'e', 'ì': 'i', 'ò': 'o', 'ù': 'u',
        'ã': 'a', 'õ': 'o',
        'ç': 'c',
        'Á': 'A', 'É': 'E', 'Í': 'I', 'Ó': 'O', 'Ú': 'U',
        'Â': 'A', 'Ê': 'E', 'Î': 'I', 'Ô': 'O', 'Û': 'U',
        'À': 'A', 'È': 'E', 'Ì': 'I', 'Ò': 'O', 'Ù': 'U',
        'Ã': 'A', 'Õ': 'O',
        'Ç': 'C'
    }

    # Remove acentos e caracteres especiais
    texto_sem_acentos = ''.join(mapeamento_acentos.get(c, c) for c in texto)
    texto_sem_caracteres_especiais = ''.join(c for c in texto_sem_acentos if c.isalnum() or c.isspace())

    return texto_sem_caracteres_especiais

def create_service():

    SERVICE_NAME = "api_socket"
    SCRIPT_PATH = "/var/python/socket/app.py"
    SERVICE_CONTENT = f"""[Unit]
                        Description=Inicializa socket
                        After=syslog.target network.target

                        [Service]
                        Type=simple
                        Restart=alwaysRestartSec=1StartLimitInterval=0
                        ExecStart=/usr/bin/python3 {SCRIPT_PATH}
                        WorkingDirectory={os.path.dirname(SCRIPT_PATH)}/
                        User=ap-002
                        StandardOutput=journal
                        StandardError=journal

                        [Install]
                        WantedBy=multi-user.target
                    """

# Caminho para o arquivo de unidade do systemd
    SERVICE_FILE = f"/var/python/socket/{SERVICE_NAME}.service"



    if os.path.exists(SERVICE_FILE):
        print("O serviço já existe.")
        return
    
    with open(SERVICE_FILE, "w") as f:
        f.write(SERVICE_CONTENT)
        f.flush()
        os.fsync(f.fileno())
    move_command = f"mv {SERVICE_FILE} /etc/systemd/system/"
    sudo_command = f"echo aro@123 | sudo -S {move_command}"
    subprocess.run(sudo_command, shell=True, check=True, input=b"aro@123\n")
           
  
    os.system("sudo systemctl daemon-reload")
    os.system(f"sudo systemctl enable {SERVICE_NAME}")
    os.system(f"sudo systemctl start {SERVICE_NAME}")
    
    print(f"Serviço {SERVICE_NAME} criado e iniciado com sucesso.")
